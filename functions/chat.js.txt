export async function onRequestPost(context) {
  try {
    const { request, env } = context;
    const body = await request.json();
    const question = body.question || "";

    if (!question.trim()) {
      return new Response(
        JSON.stringify({ error: "質問が空です" }),
        {
          status: 400,
          headers: { "Content-Type": "application/json; charset=utf-8" }
        }
      );
    }

    const groqRes = await fetch("https://api.groq.com/openai/v1/chat/completions", {
      method: "POST",
      headers: {
        "Authorization": `Bearer ${env.GROQ_API_KEY}`,
        "Content-Type": "application/json"
      },
      body: JSON.stringify({
        model: "llama-3.1-8b-instant",
        messages: [
          {
            role: "system",
            content: "あなたは社内ナレッジAIです。質問に対して、簡潔で分かりやすい日本語で回答してください。分からない場合は、分からないと正直に答えてください。"
          },
          {
            role: "user",
            content: question
          }
        ],
        temperature: 0.2
      })
    });

    const groqData = await groqRes.json();

    if (!groqRes.ok) {
      return new Response(
        JSON.stringify({
          error: "Groq APIエラー",
          detail: groqData
        }),
        {
          status: 500,
          headers: { "Content-Type": "application/json; charset=utf-8" }
        }
      );
    }

    const answer =
      groqData?.choices?.[0]?.message?.content ||
      "回答を取得できませんでした。";

    return new Response(
      JSON.stringify({ answer }),
      {
        status: 200,
        headers: { "Content-Type": "application/json; charset=utf-8" }
      }
    );
  } catch (error) {
    return new Response(
      JSON.stringify({
        error: "サーバーエラー",
        detail: String(error)
      }),
      {
        status: 500,
        headers: { "Content-Type": "application/json; charset=utf-8" }
      }
    );
  }
}
